#include <LGFX.hpp>

LGFX lcd;

void setup(void)
{
  lcd.init();
  lcd.setBrightness(255);
}

static int colors[] = {TFT_RED, TFT_GREEN, TFT_BLUE, TFT_CYAN, TFT_MAGENTA, TFT_YELLOW};

int i = 0;

void loop(void)
{
  lcd.fillRect(0, 0, 320, 320, colors[i++]);

  if (i > 5)
    i = 0;

  delay(2000);
}
